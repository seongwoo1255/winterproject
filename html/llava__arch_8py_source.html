83 def initialize_vision_modules(self, model_args, fsdp=None):
84 # 비전 타워와 관련된 구성 설정 및 초기화
85 vision_tower = model_args.vision_tower
86 mm_vision_select_layer = model_args.mm_vision_select_layer
87 mm_vision_select_feature = model_args.mm_vision_select_feature
88 pretrain_mm_mlp_adapter = model_args.pretrain_mm_mlp_adapter
89 mm_patch_merge_type = model_args.mm_patch_merge_type

91 self.config.mm_vision_tower = vision_tower

102 else:

112 self.config.mm_hidden_size = vision_tower.hidden_size
113 self.config.mm_vision_select_layer = mm_vision_select_layer
114 self.config.mm_vision_select_feature = mm_vision_select_feature
115 self.config.mm_patch_merge_type = mm_patch_merge_type

117 if getattr(self, 'mm_projector', None) is None:

121 embed_std = 1 / torch.sqrt(torch.tensor(self.config.hidden_size, dtype=self.dtype))

144def unpad_image(tensor, original_size):
145 """
146 Unpads a PyTorch tensor of a padded and resized image.
147
148 Args:
149 tensor (torch.Tensor): The image tensor, assumed to be in CxHxW format.
150 original_size (tuple): The original size of PIL image (width, height).
151
152 Returns:
153 torch.Tensor: The unpadded image tensor.
154 """
155 original_width, original_height = original_size
156 current_height, current_width = tensor.shape[1:]
157
158 original_aspect_ratio = original_width / original_height
159 current_aspect_ratio = current_width / current_height
160
161 if original_aspect_ratio > current_aspect_ratio:
162 scale_factor = current_width / original_width
163 new_height = int(original_height * scale_factor)
164 padding = (current_height - new_height) // 2
165 unpadded_tensor = tensor[:, padding:current_height - padding, :]
166 else:
167 scale_factor = current_height / original_height
168 new_width = int(original_width * scale_factor)
169 padding = (current_width - new_width) // 2
170 unpadded_tensor = tensor[:, :, padding:current_width - padding]
171
172 return unpadded_tensor

230 def prepare_inputs_labels_for_multimodal(
231 self, input_ids, position_ids, attention_mask, past_key_values, labels,
232 images, image_sizes=None
233 ):
234 vision_tower = self.get_vision_tower()
235 if vision_tower is None or images is None or input_ids.shape[1] == 1:
236 return input_ids, position_ids, attention_mask, past_key_values, None, labels
237
238 # 이미지가 여러 개일 경우 처리
239 if type(images) is list or images.ndim == 5:
240 if type(images) is list:
241 images = [x.unsqueeze(0) if x.ndim == 3 else x for x in images]
242 concat_images = torch.cat([image for image in images], dim=0)
243 image_features = self.encode_images(concat_images)
244 split_sizes = [image.shape[0] for image in images]
245 image_features = torch.split(image_features, split_sizes, dim=0)
246 # 멀티모달 패치 병합 타입 처리
247 mm_patch_merge_type = getattr(self.config, 'mm_patch_merge_type', 'flat')
248 image_aspect_ratio = getattr(self.config, 'image_aspect_ratio', 'square')
249 if mm_patch_merge_type == 'flat':
250 image_features = [x.flatten(0, 1) for x in image_features]
251 elif mm_patch_merge_type.startswith('spatial'):
252 new_image_features = []
253 for image_idx, image_feature in enumerate(image_features):
254 if image_feature.shape[0] > 1:
255 base_image_feature = image_feature[0]
256 image_feature = image_feature[1:]
257 height = width = self.get_vision_tower().num_patches_per_side
258 assert height * width == base_image_feature.shape[0]
259 if image_aspect_ratio == 'anyres':
260 num_patch_width, num_patch_height = get_anyres_image_grid_shape(image_sizes[image_idx], self.config.image_grid_pinpoints, self.get_vision_tower().config.image_size)
261 image_feature = image_feature.view(num_patch_height, num_patch_width, height, width, -1)
262 else:
263 raise NotImplementedError
264 if 'unpad' in mm_patch_merge_type:
265 image_feature = image_feature.permute(4, 0, 2, 1, 3).contiguous()
266 image_feature = image_feature.flatten(1, 2).flatten(2, 3)
267 image_feature = unpad_image(image_feature, image_sizes[image_idx])
268 image_feature = torch.cat((
269 image_feature,
270 self.model.image_newline[:, None, None].expand(*image_feature.shape[:-1], 1).to(image_feature.device)
271 ), dim=-1)
272 image_feature = image_feature.flatten(1, 2).transpose(0, 1)
273 else:
274 image_feature = image_feature.permute(0, 2, 1, 3, 4).contiguous()
275 image_feature = image_feature.flatten(0, 3)
276 image_feature = torch.cat((base_image_feature, image_feature), dim=0)
277 else:
278 image_feature = image_feature[0]
279 if 'unpad' in mm_patch_merge_type:
280 image_feature = torch.cat((
281 image_feature,
282 self.model.image_newline[None].to(image_feature.device)
283 ), dim=0)
284 new_image_features.append(image_feature)
285 image_features = new_image_features
286 else:
287 raise ValueError(f"Unexpected mm_patch_merge_type: {self.config.mm_patch_merge_type}")
288 else:
289 image_features = self.encode_images(images)
290
291 # TODO: image start / end is not implemented here to support pretraining.
292 if getattr(self.config, 'tune_mm_mlp_adapter', False) and getattr(self.config, 'mm_use_im_start_end', False):
293 raise NotImplementedError
294
295 # Let's just add dummy tensors if they do not exist,
296 # it is a headache to deal with None all the time.
297 # But it is not ideal, and if you have a better idea,
298 # please open an issue / submit a PR, thanks.
299 _labels = labels
300 _position_ids = position_ids
301 _attention_mask = attention_mask
302 if attention_mask is None:
303 attention_mask = torch.ones_like(input_ids, dtype=torch.bool)
304 else:
305 attention_mask = attention_mask.bool()
306 if position_ids is None:
307 position_ids = torch.arange(0, input_ids.shape[1], dtype=torch.long, device=input_ids.device)
308 if labels is None:
309 labels = torch.full_like(input_ids, IGNORE_INDEX)
310
311 # remove the padding using attention_mask -- FIXME
312 _input_ids = input_ids
313 input_ids = [cur_input_ids[cur_attention_mask] for cur_input_ids, cur_attention_mask in zip(input_ids, attention_mask)]
314 labels = [cur_labels[cur_attention_mask] for cur_labels, cur_attention_mask in zip(labels, attention_mask)]
315
316 # 입력 데이터에 이미지 특징 추가
317 new_input_embeds = []
318 new_labels = []
319 cur_image_idx = 0
320 for batch_idx, cur_input_ids in enumerate(input_ids):
321 num_images = (cur_input_ids == IMAGE_TOKEN_INDEX).sum()
322 if num_images == 0:
323 cur_image_features = image_features[cur_image_idx]
324 cur_input_embeds_1 = self.get_model().embed_tokens(cur_input_ids)
325 cur_input_embeds = torch.cat([cur_input_embeds_1, cur_image_features[0:0]], dim=0)
326 new_input_embeds.append(cur_input_embeds)
327 new_labels.append(labels[batch_idx])
328 cur_image_idx += 1
329 continue
330
331 image_token_indices = [-1] + torch.where(cur_input_ids == IMAGE_TOKEN_INDEX)[0].tolist() + [cur_input_ids.shape[0]]
332 cur_input_ids_noim = []
333 cur_labels = labels[batch_idx]
334 cur_labels_noim = []
335 for i in range(len(image_token_indices) - 1):
336 cur_input_ids_noim.append(cur_input_ids[image_token_indices[i]+1:image_token_indices[i+1]])
337 cur_labels_noim.append(cur_labels[image_token_indices[i]+1:image_token_indices[i+1]])
338 split_sizes = [x.shape[0] for x in cur_labels_noim]
339 cur_input_embeds = self.get_model().embed_tokens(torch.cat(cur_input_ids_noim))
340 cur_input_embeds_no_im = torch.split(cur_input_embeds, split_sizes, dim=0)
341 cur_new_input_embeds = []
342 cur_new_labels = []
343
344 for i in range(num_images + 1):
345 cur_new_input_embeds.append(cur_input_embeds_no_im[i])
346 cur_new_labels.append(cur_labels_noim[i])
347 if i < num_images:
348 cur_image_features = image_features[cur_image_idx]
349 cur_image_idx += 1
350 cur_new_input_embeds.append(cur_image_features)
351 cur_new_labels.append(torch.full((cur_image_features.shape[0],), IGNORE_INDEX, device=cur_labels.device, dtype=cur_labels.dtype))
352
353 cur_new_input_embeds = [x.to(self.device) for x in cur_new_input_embeds]
354
355 cur_new_input_embeds = torch.cat(cur_new_input_embeds)
356 cur_new_labels = torch.cat(cur_new_labels)
357
358 new_input_embeds.append(cur_new_input_embeds)
359 new_labels.append(cur_new_labels)
360
361 # Truncate sequences to max length as image embeddings can make the sequence longer
362 tokenizer_model_max_length = getattr(self.config, 'tokenizer_model_max_length', None)
363 if tokenizer_model_max_length is not None:
364 new_input_embeds = [x[:tokenizer_model_max_length] for x in new_input_embeds]
365 new_labels = [x[:tokenizer_model_max_length] for x in new_labels]
366
367 # Combine them
368 max_len = max(x.shape[0] for x in new_input_embeds)
369 batch_size = len(new_input_embeds)
370
371 new_input_embeds_padded = []
372 new_labels_padded = torch.full((batch_size, max_len), IGNORE_INDEX, dtype=new_labels[0].dtype, device=new_labels[0].device)
373 attention_mask = torch.zeros((batch_size, max_len), dtype=attention_mask.dtype, device=attention_mask.device)
374 position_ids = torch.zeros((batch_size, max_len), dtype=position_ids.dtype, device=position_ids.device)
375
376 for i, (cur_new_embed, cur_new_labels) in enumerate(zip(new_input_embeds, new_labels)):
377 cur_len = cur_new_embed.shape[0]
378 if getattr(self.config, 'tokenizer_padding_side', 'right') == "left":
379 new_input_embeds_padded.append(torch.cat((
380 torch.zeros((max_len - cur_len, cur_new_embed.shape[1]), dtype=cur_new_embed.dtype, device=cur_new_embed.device),
381 cur_new_embed
382 ), dim=0))
383 if cur_len > 0:
384 new_labels_padded[i, -cur_len:] = cur_new_labels
385 attention_mask[i, -cur_len:] = True
386 position_ids[i, -cur_len:] = torch.arange(0, cur_len, dtype=position_ids.dtype, device=position_ids.device)
387 else:
388 new_input_embeds_padded.append(torch.cat((
389 cur_new_embed,
390 torch.zeros((max_len - cur_len, cur_new_embed.shape[1]), dtype=cur_new_embed.dtype, device=cur_new_embed.device)
391 ), dim=0))
392 if cur_len > 0:
393 new_labels_padded[i, :cur_len] = cur_new_labels
394 attention_mask[i, :cur_len] = True
395 position_ids[i, :cur_len] = torch.arange(0, cur_len, dtype=position_ids.dtype, device=position_ids.device)
396
397 new_input_embeds = torch.stack(new_input_embeds_padded, dim=0)
398
399 if _labels is None:
400 new_labels = None
401 else:
402 new_labels = new_labels_padded
403
404 if _attention_mask is None:
405 attention_mask = None
406 else:
407 attention_mask = attention_mask.to(dtype=_attention_mask.dtype)
408
409 if _position_ids is None:
410 position_ids = None
411 # 결과 데이터 반환
412 return None, position_ids, attention_mask, past_key_values, new_input_embeds, new_labels

420 def initialize_vision_tokenizer(self, model_args, tokenizer):
421 if model_args.mm_use_im_patch_token:
422 tokenizer.add_tokens([DEFAULT_IMAGE_PATCH_TOKEN], special_tokens=True)
423 self.resize_token_embeddings(len(tokenizer))
424
425 if model_args.mm_use_im_start_end:
426 num_new_tokens = tokenizer.add_tokens([DEFAULT_IM_START_TOKEN, DEFAULT_IM_END_TOKEN], special_tokens=True)
427 self.resize_token_embeddings(len(tokenizer))
428
429 if num_new_tokens > 0:
430 input_embeddings = self.get_input_embeddings().weight.data
431 output_embeddings = self.get_output_embeddings().weight.data
432
433 input_embeddings_avg = input_embeddings[:-num_new_tokens].mean(
434 dim=0, keepdim=True)
435 output_embeddings_avg = output_embeddings[:-num_new_tokens].mean(
436 dim=0, keepdim=True)
437
438 input_embeddings[-num_new_tokens:] = input_embeddings_avg
439 output_embeddings[-num_new_tokens:] = output_embeddings_avg
440
441 if model_args.tune_mm_mlp_adapter:
442 for p in self.get_input_embeddings().parameters():
443 p.requires_grad = True
444 for p in self.get_output_embeddings().parameters():
445 p.requires_grad = False
446
447 if model_args.pretrain_mm_mlp_adapter:
448 mm_projector_weights = torch.load(model_args.pretrain_mm_mlp_adapter, map_location='cpu')
449 embed_tokens_weight = mm_projector_weights['model.embed_tokens.weight']
450 assert num_new_tokens == 2
451 if input_embeddings.shape == embed_tokens_weight.shape:
452 input_embeddings[-num_new_tokens:] = embed_tokens_weight[-num_new_tokens:]
453 elif embed_tokens_weight.shape[0] == num_new_tokens:
454 input_embeddings[-num_new_tokens:] = embed_tokens_weight
455 else:
456 raise ValueError(f"Unexpected embed_tokens_weight shape. Pretrained: {embed_tokens_weight.shape}. Current: {input_embeddings.shape}. Numer of new tokens: {num_new_tokens}.")
457 elif model_args.mm_use_im_patch_token:
458 if model_args.tune_mm_mlp_adapter:
459 for p in self.get_input_embeddings().parameters():
460 p.requires_grad = False
461 for p in self.get_output_embeddings().parameters():
462 p.requires_grad = False
