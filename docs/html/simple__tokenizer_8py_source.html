30@lru_cache()
31def bytes_to_unicode():
32 """
33 Returns list of utf-8 byte and a corresponding list of unicode strings.
34 The reversible bpe codes work on unicode strings.
35 This means you need a large # of unicode characters in your vocab if you want to avoid UNKs.
36 When you're at something like a 10B token dataset you end up needing around 5K for decent coverage.
37 This is a signficant percentage of your normal, say, 32K bpe vocab.
38 To avoid that, we want lookup tables between utf-8 bytes and unicode strings.
39 And avoids mapping to whitespace/control characters the bpe code barfs on.
40 """
41 bs = list(range(ord("!"), ord("~")+1))+list(range(ord("¡"), ord("¬")+1))+list(range(ord("®"), ord("ÿ")+1))
42 cs = bs[:]
43 n = 0
44 for b in range(2**8):
45 if b not in bs:
46 bs.append(b)
47 cs.append(2**8+n)
48 n += 1
49 cs = [chr(n) for n in cs]
50 return dict(zip(bs, cs))

56def get_pairs(word):
57 """Return set of symbol pairs in a word.
58 Word is represented as tuple of symbols (symbols being variable-length strings).
59 """
60 pairs = set()
61 prev_char = word[0]
62 for char in word[1:]:
63 pairs.add((prev_char, char))
64 prev_char = char
65 return pairs
