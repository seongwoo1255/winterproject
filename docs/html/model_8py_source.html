19 expansion = 4

25 def __init__(self, inplanes, planes, stride=1):
26 super().__init__()
27
28 # all conv layers have stride 1. an avgpool is performed after the second convolution when stride > 1
29 self.conv1 = nn.Conv2d(inplanes, planes, 1, bias=False)
30 self.bn1 = nn.BatchNorm2d(planes)
31 self.relu1 = nn.ReLU(inplace=True)
32
33 self.conv2 = nn.Conv2d(planes, planes, 3, padding=1, bias=False)
34 self.bn2 = nn.BatchNorm2d(planes)
35 self.relu2 = nn.ReLU(inplace=True)
36
37 self.avgpool = nn.AvgPool2d(stride) if stride > 1 else nn.Identity()
38
39 self.conv3 = nn.Conv2d(planes, planes * self.expansion, 1, bias=False)
40 self.bn3 = nn.BatchNorm2d(planes * self.expansion)
41 self.relu3 = nn.ReLU(inplace=True)
42
43 self.downsample = None
44 self.stride = stride
45
46 if stride > 1 or inplanes != planes * Bottleneck.expansion:
47 # downsampling layer is prepended with an avgpool, and the subsequent convolution has stride 1
48 self.downsample = nn.Sequential(OrderedDict([
49 ("-1", nn.AvgPool2d(stride)),
50 ("0", nn.Conv2d(inplanes, planes * self.expansion, 1, stride=1, bias=False)),
51 ("1", nn.BatchNorm2d(planes * self.expansion))
52 ]))
53
54
57 def forward(self, x: torch.Tensor):
58 identity = x
59
60 out = self.relu1(self.bn1(self.conv1(x)))
61 out = self.relu2(self.bn2(self.conv2(out)))
62 out = self.avgpool(out)
63 out = self.bn3(self.conv3(out))
64
65 if self.downsample is not None:
66 identity = self.downsample(x)
67
68 out += identity
69 out = self.relu3(out)
70 return out

85 def __init__(self, spacial_dim: int, embed_dim: int, num_heads: int, output_dim: int = None):
86 super().__init__()

88 self.k_proj = nn.Linear(embed_dim, embed_dim)
89 self.q_proj = nn.Linear(embed_dim, embed_dim)

94
97 def forward(self, x):
98 x = x.flatten(start_dim=2).permute(2, 0, 1) # NCHW -> (HW)NC
99 x = torch.cat([x.mean(dim=0, keepdim=True), x], dim=0) # (HW+1)NC
100 x = x + self.positional_embedding[:, None, :].to(x.dtype) # (HW+1)NC
101 x, _ = F.multi_head_attention_forward(
102 query=x[:1], key=x, value=x,
103 embed_dim_to_check=x.shape[-1],
104 num_heads=self.num_heads,
105 q_proj_weight=self.q_proj.weight,
106 k_proj_weight=self.k_proj.weight,
107 v_proj_weight=self.v_proj.weight,

109 in_proj_bias=torch.cat([self.q_proj.bias, self.k_proj.bias, self.v_proj.bias]),

112 add_zero_attn=False,
113 dropout_p=0,

130class ModifiedResNet(nn.Module):
131 """
132 A ResNet class that is similar to torchvision's but contains the following changes:
133 - There are now 3 "stem" convolutions as opposed to 1, with an average pool instead of a max pool.
134 - Performs anti-aliasing strided convolutions, where an avgpool is prepended to convolutions with stride > 1
135 - The final pooling layer is a QKV attention instead of an average pool
136 """
137
143 def __init__(self, layers, output_dim, heads, input_resolution=224, width=64):
144 super().__init__()
145 self.output_dim = output_dim
146 self.input_resolution = input_resolution
147
148 # the 3-layer stem
149 self.conv1 = nn.Conv2d(3, width // 2, kernel_size=3, stride=2, padding=1, bias=False)
150 self.bn1 = nn.BatchNorm2d(width // 2)
151 self.relu1 = nn.ReLU(inplace=True)
152 self.conv2 = nn.Conv2d(width // 2, width // 2, kernel_size=3, padding=1, bias=False)
153 self.bn2 = nn.BatchNorm2d(width // 2)
154 self.relu2 = nn.ReLU(inplace=True)
155 self.conv3 = nn.Conv2d(width // 2, width, kernel_size=3, padding=1, bias=False)
156 self.bn3 = nn.BatchNorm2d(width)
157 self.relu3 = nn.ReLU(inplace=True)
158 self.avgpool = nn.AvgPool2d(2)
159
160 # residual layers
161 self._inplanes = width # this is a *mutable* variable used during construction
162 self.layer1 = self._make_layer(width, layers[0])
163 self.layer2 = self._make_layer(width * 2, layers[1], stride=2)
164 self.layer3 = self._make_layer(width * 4, layers[2], stride=2)
165 self.layer4 = self._make_layer(width * 8, layers[3], stride=2)
166
167 embed_dim = width * 32 # the ResNet feature dimension
168 self.attnpool = AttentionPool2d(input_resolution // 32, embed_dim, heads, output_dim)

175 def _make_layer(self, planes, blocks, stride=1):
176 layers = [Bottleneck(self._inplanes, planes, stride)]
177
178 self._inplanes = planes * Bottleneck.expansion
179 for _ in range(1, blocks):
180 layers.append(Bottleneck(self._inplanes, planes))

182 return nn.Sequential(*layers)

299class VisionTransformer(nn.Module):
300
307 def __init__(self, input_resolution: int, patch_size: int, width: int, layers: int, heads: int, output_dim: int):
308 super().__init__()
309 self.input_resolution = input_resolution
310 self.output_dim = output_dim
311 self.conv1 = nn.Conv2d(in_channels=3, out_channels=width, kernel_size=patch_size, stride=patch_size, bias=False)
312
313 scale = width ** -0.5
314 self.class_embedding = nn.Parameter(scale * torch.randn(width))
315 self.positional_embedding = nn.Parameter(scale * torch.randn((input_resolution // patch_size) ** 2 + 1, width))
316 self.ln_pre = LayerNorm(width)
317
318 self.transformer = Transformer(width, layers, heads)
319
320 self.ln_post = LayerNorm(width)
321 self.proj = nn.Parameter(scale * torch.randn(width, output_dim))
322
323
326 def forward(self, x: torch.Tensor):
327 x = self.conv1(x) # shape = [*, width, grid, grid]
328 x = x.reshape(x.shape[0], x.shape[1], -1) # shape = [*, width, grid ** 2]
329 x = x.permute(0, 2, 1) # shape = [*, grid ** 2, width]
330 x = torch.cat([self.class_embedding.to(x.dtype) + torch.zeros(x.shape[0], 1, x.shape[-1], dtype=x.dtype, device=x.device), x], dim=1) # shape = [*, grid ** 2 + 1, width]
331 x = x + self.positional_embedding.to(x.dtype)
332 x = self.ln_pre(x)
333
334 x = x.permute(1, 0, 2) # NLD -> LND
335 x = self.transformer(x)
336 x = x.permute(1, 0, 2) # LND -> NLD
337
338 x = self.ln_post(x[:, 0, :])
339
340 if self.proj is not None:
341 x = x @ self.proj
342
343 return x

416 def initialize_parameters(self):
417 nn.init.normal_(self.token_embedding.weight, std=0.02)
418 nn.init.normal_(self.positional_embedding, std=0.01)

447 def build_attention_mask(self):
448 # lazily create causal attention mask, with full attention between the vision tokens
449 # pytorch uses additive attention mask; fill with -inf
450 mask = torch.empty(self.context_length, self.context_length)
451 mask.fill_(float("-inf"))
452 mask.triu_(1) # zero out the lower diagonal
453 return mask

470 def encode_text(self, text):
471 x = self.token_embedding(text).type(self.dtype) # [batch_size, n_ctx, d_model]
472
473 x = x + self.positional_embedding.type(self.dtype)
474 x = x.permute(1, 0, 2) # NLD -> LND
475 x = self.transformer(x)
476 x = x.permute(1, 0, 2) # LND -> NLD
477 x = self.ln_final(x).type(self.dtype)
478
479 # x.shape = [batch_size, n_ctx, transformer.width]
480 # take features from the eot embedding (eot_token is the highest number in each sequence)
481 x = x[torch.arange(x.shape[0]), text.argmax(dim=-1)] @ self.text_projection
482
483 return x

489 def forward(self, image, text):
490 image_features = self.encode_image(image)
491 text_features = self.encode_text(text)
492
493 # normalized features
494 image_features = image_features / image_features.norm(dim=1, keepdim=True)
495 text_features = text_features / text_features.norm(dim=1, keepdim=True)
496
497 # cosine similarity as logits
498 logit_scale = self.logit_scale.exp()
499 logits_per_image = logit_scale * image_features @ text_features.t()
500 logits_per_text = logits_per_image.t()
501
502 # shape = [global_batch_size, global_batch_size]
503 return logits_per_image, logits_per_text

511
512 def _convert_weights_to_fp16(l):
513 if isinstance(l, (nn.Conv1d, nn.Conv2d, nn.Linear)):
514 l.weight.data = l.weight.data.half()
515 if l.bias is not None:
516 l.bias.data = l.bias.data.half()
517
518 if isinstance(l, nn.MultiheadAttention):
519 for attr in [*[f"{s}_proj_weight" for s in ["in", "q", "k", "v"]], "in_proj_bias", "bias_k", "bias_v"]:
520 tensor = getattr(l, attr)
521 if tensor is not None:
522 tensor.data = tensor.data.half()
523
524 for name in ["text_projection", "proj"]:
525 if hasattr(l, name):
526 attr = getattr(l, name)
527 if attr is not None:
528 attr.data = attr.data.half()
