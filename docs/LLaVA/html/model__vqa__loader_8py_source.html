19def split_list(lst, n):
20 """Split a list into n (roughly) equal-sized chunks"""
21 chunk_size = math.ceil(len(lst) / n) # integer division
22 return [lst[i:i+chunk_size] for i in range(0, len(lst), chunk_size)]
